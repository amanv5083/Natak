package com.example.myapplication

import androidx.appcompat.app.AppCompatActivity
import android.os.Bundle
import android.widget.Button
import android.widget.EditText

class Calculator : AppCompatActivity() {

    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        setContentView(R.layout.activity_calculator)

        val etNum1 = findViewById<EditText>(R.id.txtno1)
        val etNum2 = findViewById<EditText>(R.id.txtno2)
        val btnAdd = findViewById<Button>(R.id.btnAdd)
        val btnSub = findViewById<Button>(R.id.btnSub)
        val btnMul = findViewById<Button>(R.id.btnMul)
        val btnDiv = findViewById<Button>(R.id.btnDiv)
        val txtresult = findViewById<EditText>(R.id.txtresult)

        btnAdd.setOnClickListener {
            val num1 = etNum1.text.toString().toIntOrNull() ?: 0
            val num2 = etNum2.text.toString().toIntOrNull() ?: 0
            txtresult.setText((num1 + num2).toString())
        }

        btnSub.setOnClickListener {
            val num1 = etNum1.text.toString().toIntOrNull() ?: 0
            val num2 = etNum2.text.toString().toIntOrNull() ?: 0
            txtresult.setText((num1 - num2).toString())
        }

        btnMul.setOnClickListener {
            val num1 = etNum1.text.toString().toIntOrNull() ?: 0
            val num2 = etNum2.text.toString().toIntOrNull() ?: 0
            txtresult.setText((num1 * num2).toString())
        }

        btnDiv.setOnClickListener {
            val num1 = etNum1.text.toString().toIntOrNull() ?: 0
            val num2 = etNum2.text.toString().toIntOrNull() ?: 1

            if (num2 != 0) {
                txtresult.setText((num1 / num2).toString())
            } else {
                txtresult.setText("Cannot divide by 0")
            }
        }
    }
}
