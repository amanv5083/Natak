Practical 2 - Basic Calculator

Practical No. 02

Aim: Write a Kotlin program to display basic calculator (ADD, MUL, SUB, DIV).

fun main() {

    println("Enter first number:")
    val num1 = 45
    println(num1)

    println("==========================")

    println("Enter second number:")
    val num2 = 9
    println(num2)

    println("============= Basic Calculator =============")

    val sum = num1 + num2
    println("The sum is: $sum")

    val m = num1 * num2
    println("The Multiply is: $m")

    val s = num1 - num2
    println("The Subtract is: $s")

    val dd = num1 / num2
    println("The div is: $dd")
}
