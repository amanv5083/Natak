Practical No: 2

Aim: Write a Kotlin program to display Basic Calculator (ADD, MUL, DIV, SUB).

fun main() {

    println("Enter first number:")
    val num1 = 500
    println(num1)
    println("______")

    println("Enter second number:")
    val num2 = 10
    println(num2)

    println("____ Basic Calculator ____")

    val sum = num1 + num2
    println("The Sum is: $sum")

    val multiply = num1 * num2
    println("The Multiply is: $multiply")

    val subtract = num1 - num2
    println("The Subtract is: $subtract")

    val division = num1 / num2
    println("The Division is: $division")
}

Practical No: 3

Aim: Write a Kotlin program to display 1 to 10 multiplication table.

fun main() {

    for (i in 1..10) {
        for (j in 1..10) {
            print("${i * j}\t")
        }
        println()
    }

}

fun main() {
